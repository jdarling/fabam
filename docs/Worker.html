Example

Simple worker that prepends "echo: " to whatever it is sent and then sends it back to the manager

const {
  Worker
} = require('fabam');

const worker = new Worker();

worker.on('echo', (what, callback)=>{
  callback(`echo: ${what}`);
});

worker.ready();
